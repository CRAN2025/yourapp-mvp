Add this tiny helper component near the top of StorefrontPublicView.jsx (below imports is fine):

// --- Full-bleed section helper (edge-to-edge band) ---
const FullBleedSection = ({ children }) => (
  <section
    style={{
      width: '100vw',
      position: 'relative',
      left: '50%',
      right: '50%',
      marginLeft: '-50vw',
      marginRight: '-50vw',
      // pick your hero gradient; the band itself is full width
      background: 'linear-gradient(135deg,#6f3ef4 0%,#38bdf8 100%)',
      padding: '24px 0',
    }}
  >
    <div
      style={{
        maxWidth: 980,
        margin: '0 auto',
        paddingInline: 'clamp(12px,4vw,24px)',
      }}
    >
      {children}
    </div>
  </section>
);
